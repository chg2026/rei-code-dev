━━━ FLIP MATH ━━━
totalHoldMonths = Number(rehabMonths) + Number(monthsToSell)
totalHoldingCost = monthlyHolding * totalHoldMonths
costOfSale = arv * (costOfSalePct / 100)
allInExclSale = purchasePrice + closingCosts + totalHoldingCost + rehabCost
cashRequired = financingType === 'cash' ? allInExclSale : allInExclSale - loanAmount
profit = arv - allInExclSale - costOfSale
roi = cashRequired > 0 ? (profit / cashRequired) * 100 : 0
roiAnnualized = totalHoldMonths > 0 ? (Math.pow(1 + roi/100, 12/totalHoldMonths) - 1) * 100 : 0
equityMultiple = cashRequired > 0 ? (cashRequired + profit) / cashRequired : 0

━━━ BRRRR MATH ━━━
refiLoan = arv * (refinanceLTV / 100)
r = refinanceRate / 100 / 12
n = Number(refinanceTerm)
monthlyRefiPayment = r > 0 ? refiLoan * r / (1 - Math.pow(1+r, -n)) : refiLoan / n
effectiveRent = monthlyRent * (1 - vacancyPct / 100)
monthlyCF = effectiveRent - monthlyExpenses - monthlyRefiPayment
annualCF = monthlyCF * 12
totalInvested = purchasePrice + closingCosts + rehabCost + (monthlyHolding * Number(rehabMonths))
cashLeft = totalInvested - refiLoan
cashOnCash = cashLeft > 0 ? (annualCF / cashLeft) * 100 : 0
equityAtAcq = arv - refiLoan

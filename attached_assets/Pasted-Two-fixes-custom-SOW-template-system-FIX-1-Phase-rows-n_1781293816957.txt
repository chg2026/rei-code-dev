STEP A — Schema (raw SQL + prisma generate, NOT prisma db push)
Run migration at apps/chg-rehab/scripts/migrate-sow-templates.ts:

CREATE TABLE "SowTemplate" (
  id TEXT PRIMARY KEY DEFAULT gen_random_uuid()::text,
  "companyId" TEXT NOT NULL,
  name TEXT NOT NULL,
  description TEXT,
  "createdById" TEXT NOT NULL,
  "createdAt" TIMESTAMPTZ NOT NULL DEFAULT now(),
  "updatedAt" TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE "SowTemplatePhase" (
  id TEXT PRIMARY KEY DEFAULT gen_random_uuid()::text,
  "templateId" TEXT NOT NULL REFERENCES "SowTemplate"(id) ON DELETE CASCADE,
  number INT NOT NULL,
  name TEXT NOT NULL,
  description TEXT,
  "laborBudget" DECIMAL(14,2) NOT NULL DEFAULT 0,
  "materialsBudget" DECIMAL(14,2) NOT NULL DEFAULT 0,
  dependencies INT[] NOT NULL DEFAULT '{}',
  "acceptanceCriteria" TEXT[] NOT NULL DEFAULT '{}',
  UNIQUE("templateId", number)
);

Add to schema.prisma:

model SowTemplate {
  id            String             @id @default(cuid())
  companyId     String
  name          String
  description   String?
  createdById   String
  createdAt     DateTime           @default(now())
  updatedAt     DateTime           @updatedAt
  phases        SowTemplatePhase[]
}

model SowTemplatePhase {
  id                 String      @id @default(cuid())
  templateId         String
  number             Int
  name               String
  description        String?
  laborBudget        Decimal     @default(0) @db.Decimal(14, 2)
  materialsBudget    Decimal     @default(0) @db.Decimal(14, 2)
  dependencies       Int[]
  acceptanceCriteria String[]
  template           SowTemplate @relation(fields: [templateId], references: [id], onDelete: Cascade)

  @@unique([templateId, number])
}

Run: npx prisma generate

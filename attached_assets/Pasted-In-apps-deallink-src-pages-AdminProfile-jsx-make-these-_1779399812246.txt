B) Inside the AdminProfile component, after the existing state declarations, add:
  const fileInputRef = React.useRef(null);
  const [uploading, setUploading] = React.useState(false);

  async function handlePhotoUpload(e) {
    const file = e.target.files?.[0];
    if (!file) return;
    setUploading(true);
    try {
      const ext = file.name.split('.').pop() || 'jpg';
      const path = `profiles/${form.handle || 'user'}/avatar-${Date.now()}.${ext}`;
      const { error } = await supabase.storage
        .from('deal-photos')
        .upload(path, file, { upsert: true, contentType: file.type });
      if (error) throw error;
      const { data: urlData } = supabase.storage.from('deal-photos').getPublicUrl(path);
      setField('avatarUrl', urlData.publicUrl);
      show('Photo uploaded');
    } catch (err) {
      show(err?.message || 'Upload failed');
    } finally {
      setUploading(false);
      e.target.value = '';
    }
  }

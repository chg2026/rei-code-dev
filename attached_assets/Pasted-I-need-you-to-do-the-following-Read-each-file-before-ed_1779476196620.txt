```jsx
import React, { useEffect, useRef } from 'react';

export default function FlipBrrrrCalc({ deal, dispatch }) {
  const iframeRef = useRef(null);

  // Pre-fill the calculator when the iframe loads
  function handleLoad() {
    const frame = iframeRef.current;
    if (!frame) return;
    try {
      frame.contentWindow.postMessage({
        type: 'REI_PREFILL',
        deal: {
          purchasePrice: deal?.purchase_price || 0,
          rehabCost: deal?.rehab_budget || 0,
          arv: deal?.arv || 0,
          address: deal?.address || '',
          calcState: deal?.imConfig?.calcState || null,
        }
      }, '*');
    } catch (e) {}
  }

  // Listen for save messages from the iframe
  useEffect(() => {
    function handleMessage(e) {
      if (e.data?.type === 'REI_SAVE' && e.data?.calcState) {
        dispatch({
          type: 'update_deal',
          id: deal.id,
          patch: { imConfig: { ...deal.imConfig, calcState: e.data.calcState } }
        });
      }
    }
    window.addEventListener('message', handleMessage);
    return () => window.removeEventListener('message', handleMessage);
  }, [deal, dispatch]);

  return (
    <div style={{ width: '100%', height: '100vh', minHeight: 700 }}>
      <iframe
        ref={iframeRef}
        src="/deal-calc.html"
        onLoad={handleLoad}
        style={{ width: '100%', height: '100%', border: 'none', display: 'block' }}
        title="Deal Calculator"
      />
    </div>
  );
}
```

```html
<script>
// REI Flywheel ↔ Calculator bridge
window.addEventListener('message', function(e) {
  if (e.data && e.data.type === 'REI_PREFILL') {
    var d = e.data.deal || {};
    // Restore saved state if present
    if (d.calcState) {
      try { restoreCalcState(d.calcState); } catch(err) {}
    }
    // Pre-fill assumption panel inputs
    if (d.purchasePrice) {
      var pp = document.getElementById('inp-purchase');
      if (pp) { pp.value = d.purchasePrice; pp.dispatchEvent(new Event('input')); }
    }
    if (d.rehabCost) {
      var rc = document.getElementById('inp-rehab');
      if (rc) { rc.value = d.rehabCost; rc.dispatchEvent(new Event('input')); }
    }
    if (d.arv) {
      var arvEl = document.getElementById('inp-arv');
      if (arvEl) { arvEl.value = d.arv; arvEl.dispatchEvent(new Event('input')); }
    }
  }
});

function getCalcState() {
  // Collect all input values from the assumption panel
  var inputs = {};
  document.querySelectorAll('[id^="inp-"]').forEach(function(el) {
    inputs[el.id] = el.value;
  });
  return inputs;
}

function restoreCalcState(state) {
  Object.keys(state).forEach(function(id) {
    var el = document.getElementById(id);
    if (el) { el.value = state[id]; el.dispatchEvent(new Event('input')); }
  });
}

// Auto-save every 5 seconds
setInterval(function() {
  try {
    window.parent.postMessage({ type: 'REI_SAVE', calcState: getCalcState() }, '*');
  } catch(e) {}
}, 5000);
</script>
```
